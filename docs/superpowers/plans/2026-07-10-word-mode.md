> **For agentic workers:** REQUIRED SUB-SKILL: Use superpowers:subagent-driven-development (recommended) or superpowers:executing-plans to implement this plan task-by-task. Steps use checkbox (`- [ ]`) syntax for tracking.

**Goal:** Add a "Random / Words" mode toggle that lets users generate hyphen-separated passphrases (e.g. `horse-battery-staple-lamp`) instead of random character strings.

The current `password_gen.py` imports `secrets` and `string`. Add the import for `WORD_LIST` and the new function. Replace the entire file with:

```python
import secrets
import string
from words import WORD_LIST


def generate_password(length=12, use_upper=True, use_digits=True, use_symbols=True):
    characters = string.ascii_lowercase

    if use_upper:
        characters += string.ascii_uppercase

    if use_digits:
        characters += string.digits

    if use_symbols:
        characters += string.punctuation

    password = "".join(secrets.choice(characters) for _ in range(length))
    return password


def generate_passphrase(word_count=4):
    return "-".join(secrets.choice(WORD_LIST) for _ in range(word_count))
```

```python
from flask import Flask, render_template, request, jsonify
from password_gen import generate_password, generate_passphrase

app = Flask(__name__)


@app.route("/")
def index():
    return render_template("index.html")


@app.route("/generate", methods=["POST"])
def generate():
    data = request.get_json(silent=True) or {}
    mode = data.get("mode", "random")

    if mode == "words":
        word_count = data.get("word_count", 4)
        if not isinstance(word_count, int) or isinstance(word_count, bool) or not (3 <= word_count <= 10):
            return jsonify({"error": "Word count must be an integer between 3 and 10"}), 400
        return jsonify({"password": generate_passphrase(word_count)})

    length = data.get("length", 12)
    if not isinstance(length, int) or isinstance(length, bool) or not (8 <= length <= 64):
        return jsonify({"error": "Length must be an integer between 8 and 64"}), 400

    use_upper = bool(data.get("use_upper", True))
    use_digits = bool(data.get("use_digits", True))
    use_symbols = bool(data.get("use_symbols", True))

    return jsonify({"password": generate_password(length, use_upper, use_digits, use_symbols)})


if __name__ == "__main__":
    app.run(debug=True)
```

```html
  <script>
    const slider = document.getElementById('lengthSlider');
    const display = document.getElementById('lengthDisplay');
    const upperToggle = document.getElementById('upperToggle');
    const digitsToggle = document.getElementById('digitsToggle');
    const symbolsToggle = document.getElementById('symbolsToggle');
    const generateBtn = document.getElementById('generateBtn');
    const passwordOutput = document.getElementById('passwordOutput');
    const copyBtn = document.getElementById('copyBtn');
    const message = document.getElementById('message');
    const modeRandom = document.getElementById('modeRandom');
    const modeWords = document.getElementById('modeWords');
    const randomControls = document.getElementById('random-controls');
    const wordControls = document.getElementById('word-controls');
    const wordCountSlider = document.getElementById('wordCountSlider');
    const wordCountDisplay = document.getElementById('wordCountDisplay');

    let currentMode = 'random';

    // Length slider live update
    slider.addEventListener('input', () => {
      display.textContent = slider.value;
    });

    // Word count slider live update
    wordCountSlider.addEventListener('input', () => {
      wordCountDisplay.textContent = wordCountSlider.value;
    });

    // Mode toggle
    modeRandom.addEventListener('click', () => {
      currentMode = 'random';
      modeRandom.classList.add('active');
      modeWords.classList.remove('active');
      randomControls.style.display = '';
      wordControls.style.display = 'none';
      updateGenerateBtn();
    });

    modeWords.addEventListener('click', () => {
      currentMode = 'words';
      modeWords.classList.add('active');
      modeRandom.classList.remove('active');
      randomControls.style.display = 'none';
      wordControls.style.display = '';
      generateBtn.disabled = false;
      clearMessage();
    });

    // Disable generate if all random toggles are off (only applies in random mode)
    function updateGenerateBtn() {
      if (currentMode === 'words') {
        generateBtn.disabled = false;
        return;
      }
      const anyOn = upperToggle.checked || digitsToggle.checked || symbolsToggle.checked;
      generateBtn.disabled = !anyOn;
      if (!anyOn) {
        showMessage('Select at least one character type', 'error');
      } else {
        clearMessage();
      }
    }

    [upperToggle, digitsToggle, symbolsToggle].forEach(t =>
      t.addEventListener('change', updateGenerateBtn)
    );

    generateBtn.addEventListener('click', async () => {
      clearMessage();
      generateBtn.textContent = 'Generating…';
      generateBtn.disabled = true;

      const payload = currentMode === 'words'
        ? { mode: 'words', word_count: parseInt(wordCountSlider.value, 10) }
        : {
            length: parseInt(slider.value, 10),
            use_upper: upperToggle.checked,
            use_digits: digitsToggle.checked,
            use_symbols: symbolsToggle.checked,
          };

      try {
        const res = await fetch('/generate', {
          method: 'POST',
          headers: { 'Content-Type': 'application/json' },
          body: JSON.stringify(payload),
        });

        const data = await res.json();

        if (!res.ok) {
          showMessage(data.error || 'Something went wrong, try again', 'error');
        } else {
          passwordOutput.value = data.password;
        }
      } catch {
        showMessage('Something went wrong, try again', 'error');
      } finally {
        generateBtn.textContent = 'Generate Password';
        const anyOn = upperToggle.checked || digitsToggle.checked || symbolsToggle.checked;
        generateBtn.disabled = currentMode === 'words' ? false : !anyOn;
      }
    });

    copyBtn.addEventListener('click', () => {
      if (!passwordOutput.value) return;
      navigator.clipboard.writeText(passwordOutput.value).then(() => {
        copyBtn.textContent = '✓';
        setTimeout(() => { copyBtn.innerHTML = '&#128203;'; }, 1500);
      }).catch(() => {
        showMessage('Copy failed — select and copy manually', 'error');
      });
    });

    function showMessage(text, type) {
      message.textContent = text;
      message.className = type;
    }

    function clearMessage() {
      message.textContent = '';
      message.className = '';
    }
  </script>
```
